def to_feed(rss, item)
  enclosure = Rss::Channel::Item::Enclosure.new
  set = setup_values(enclosure)
  if set
    item.enclosure = enclosure
    set_parent(enclosure, item)
    setup_other_elements(rss, enclosure)
  end
end
